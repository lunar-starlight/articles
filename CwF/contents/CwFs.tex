\section{Categories with Families}

\begin{definition}
  A \emph{category with families} is comprised of the following data
  \begin{itemize}
  \item Category of contexts \(\ctx\) with a terminal object \(ε\).
  \item Presheaf of types \(\ty\) over \(\ctx\).
  \item Dependent presheaf of terms \(\tmop\) over \(\ty\).
  \item Dependent context extension functor of types to contexts \(\ext - -\).
  \item Natural isomorphism
    \(\exttm - - : \sigma{γ : \ctx(Δ,Γ)}{\tm Δ {A\subst γ}} ≅ \ctx(Δ,\ext Γ A)\)
    in \(Γ\) and \(Δ\) with inverse \(φ\).
  \end{itemize}

  We also define the projections and variable terms as \(\p{π_Γ^A, \var_Γ^A} ≔ φ(\id_{\ext Γ A})\).
  We will often omit the type and context annotations, as they can be deduced.
\end{definition}

\begin{theorem}
  Let \(γ : Δ → Γ\), \(δ : Θ → Δ\), \(A : \ty Γ\), and \(t : \tm Δ {A\subst γ}\).
  Then the following equalities hold.
  \begin{itemize}
  \item \(\exttm γ A = \exttm {γ∘π} {\var}\)
  \item \(\exttm γ t ∘ δ = \exttm {γ∘δ} {t\subst δ}\)
  \item \(π ∘ \exttm γ t = γ\)
  \item \(\var\substext γ t = t\)
  \end{itemize}
\end{theorem}
\begin{proof}
  The first equality expresses naturality of the isomorphism in \(Γ\), while the
  second expresses naturality in \(Δ\).

  For the other two equalities we can write
  \[ \exttm γ t = \id ∘ \exttm γ t = \exttm π {\var} ∘ \exttm γ t = \exttm {π ∘ \exttm γ t}
    {\var\substext γ t}\text.\qedhere \]
\end{proof}

\begin{definition}
  A category with families has dependent product types or \(Π\) types when for
  any \(A : \ty Γ\) and \(B : \ty {\ext Γ A}\) we have
  \begin{itemize}
  \item types \(Π A B : \ty Γ\) such that
    \(\p{Π A B} \subst γ = Π A\subst γ B\substext γ A\) and
  \item a natural isomorphism \(\hat{\p -} : \tm Γ {\p{Π A B}} ≅ \tm {\ext Γ A} B : λ_Γ^{Α ⊢ Β}\) in \(Γ\).
  \end{itemize}
\end{definition}
Given \(f : \tm Γ {\p{Π Α Β}}\) and \(t : \tm Γ A\) we can define application as
\[ \ap f\: t ≔ \hat f(t) : \tm Γ B(t)\text, \]
where \(-(t) ≔ -\substext \id t\).


\begin{definition}
  A category with families has dependent sum types or \(Σ\) types when for
  any \(A : \ty Γ\) and \(B : \ty {\ext Γ A}\) we have
  \begin{itemize}
  \item types \(Σ A B : \ty Γ\) such that
    \(\p{Σ A B} \subst γ = Σ A\subst γ B\substext γ A\) and
  \item a natural isomorphism
    \(\pair : \ext Γ {\p{Σ A B}} ≅ \ext {\ext Γ A} B\) in \(Γ\).
  \item (alt) a (natural?) isomorphism
    \(\pair : \sigma{a : \tm Γ A}{b : \tm Γ {B(a)}} ≅ \tm Γ {\p{Σ Α Β}}\)
  \end{itemize}
\end{definition}

\begin{definition}
  A category with families has unit types when we have a type \(\1 : \ty \emp\)
  such that \(\emp ≅ \ext \emp \1\).
\end{definition}

\begin{definition}
  A category with families has natural numbers types when we have a type
  \(ℕ : \ty \emp\) such that \(\ext \emp ℕ\) is a natural numbers object in
  \(\ctx\).
\end{definition}

\begin{definition}
  A category with families has identity types when we have
  \begin{itemize}
  \item types \(\idty[A] : \ty {\ext {\ext Γ A} A^↑}\) such that
    \(\idty[A]\subst{γ^{↑↑}} = \idty[A\subst γ]\) and
  \item a natural isomorphism \(\refl : \ext Γ A ≅ \ext {\ext {\ext Γ A} {A^↑}} {\idty[A]}\).
  \end{itemize}
\end{definition}
